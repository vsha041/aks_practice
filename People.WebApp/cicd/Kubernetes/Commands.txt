// AKS

kubectl apply -f .\people_aks.yml

// LOCAL
kubectl apply -f .\people_deployment.yml
kubectl apply -f .\people_service.yml

// DEBUGGING
kubectl run test-client --rm -it --image=curlimages/curl -- sh
kubectl port-forward svc/service-name 8080:5000

// see all pods
kubectl get pods --all-namespaces

// see all services
kubectl get svc --all-namespaces

// see pod configration
kubectl describe pod

// see logs
kubectl logs -l app=name_of_the_app

// get service
kubectl get svc service-name

// see all nodes
kubectl get nodes

// set kubernetes context for commands
 az aks get-credentials -g rg-aks-people -n aks-people

 // see history of rollouts
 kubectl rollout history deployment/people-webapp-deployment

 // revert to N - 1 version 
 kubectl rollout undo deployment/people-webapp-deployment

 // rever to version 3
 kubectl rollout undo deployment/people-webapp-deployment --to-revision=3

 // restart deployment from the beginning
 kubectl rollout restart deployment/people-webapp-deployment

 // scale number of PODS to 8 but try to use K8 Horizontal POD AutoScalar (HPA)
 kubectl scale deployment/people-webapp-deployment --replicas=8
